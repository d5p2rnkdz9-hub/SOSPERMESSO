**Implementation approach:**

Write a Node.js script (or use bash with careful regex) that:
1. Finds all HTML files in src/pages/
2. For each file, finds the nav-menu block
3. Replaces it with the dropdown template
4. Writes back to file

Example script approach:
```javascript
const fs = require('fs');
const path = require('path');
const glob = require('glob'); // or use fs.readdirSync

const DROPDOWN_TEMPLATE = `<ul class="nav-menu" id="nav-menu">
  <!-- ... full template here ... -->
</ul>`;

const files = glob.sync('src/pages/*.html');
files.forEach(file => {
  let content = fs.readFileSync(file, 'utf8');
  // Replace nav-menu block (multiline regex)
  content = content.replace(
    /<ul class="nav-menu"[^>]*>[\s\S]*?<\/ul>/,
    DROPDOWN_TEMPLATE
  );
  fs.writeFileSync(file, content);
});
```

Alternative: Use multiple Read/Write tool calls to update files in batches.
